def draw_koch(t, depth, length):  
    if depth == 1:
        t.fd(length)
    if depth > 1:
        draw_koch(t, depth - 1, length)
        t.lt(60)
        draw_koch(t, depth - 1, length)
        t.rt(120)
        draw_koch(t, depth - 1, length)
        t.lt(60)
        draw_koch(t, depth - 1, length)

def triangle(t, size):
    t.lt(60)
    t.fd(size)
    t.rt(120)
    t.fd(size)
    t.rt(120)
    t.fd(size)
    t.rt(180)
    
def pinski(t, depth, length):
  if depth == 1:
        triangle(t, length)
    if depth > 1:
        pinski(t, depth - 1, length / 2)
        t.fd(length / 2)
        pinski(t, depth - 1, length / 2)
        t.fd(-length/2 )
        t.lt(60)
        t.fd(length/2)
        t.rt(60)
        pinski(t, depth - 1, length / 2)
        t.rt(120)
        t.fd(length/2)
        t.lt(120)

def tree_mod(t, depth, size, angle):
    t.pensize(5)
    if depth == 0:
        t.color(random.randrange(255), random.randrange(255), random.randrange(255))
        t.fd(size)
    else:
        t.fd(size)
        t.rt(angle)
        tree_mod(t, depth-1, size, angle * 2)
        t.rt(2 * angle)
        tree_mod(t, depth-1, size, angle * 2)
        t.rt(angle)
        t.color(random.randrange(255), random.randrange(255), random.randrange(255))
        t.bk(size * 3)
